// 购物车数组，每个商品为对象
const cart = [];

// 添加商品函数
function addItem(name, price, quantity) {
    cart.push({
        name: name,
        price: parseFloat(price),
        quantity: parseInt(quantity)
    });
    console.log(`添加商品: ${name}, 单价: ¥${price}, 数量: ${quantity}`);
    updateCartDisplay();
}
// 计算总价函数 - 使用reduce方法
function calcTotal() {
    // 使用reduce方法计算总价，优雅且函数式
    const total = cart.reduce((accumulator, item) => {
        return accumulator + (item.price * item.quantity);
    }, 0);
    
    console.log(`计算总价: ¥${total.toFixed(2)}`);
    return total;
}

// 使用forEach的替代实现
function calcTotalWithForEach() {
    let total = 0;
    cart.forEach(item => {
        total += item.price * item.quantity;
    });
    return total;
}
// 应用优惠函数 - 接收回调函数作为参数
function applyDiscount(callback) {
    const originalPrice = calcTotal();
    console.log(`原总价: ¥${originalPrice.toFixed(2)}`);
    
    // 回调函数接收原总价，返回折后价
    const finalPrice = callback(originalPrice);
    
    console.log(`应用优惠后: ¥${finalPrice.toFixed(2)}`);
    return finalPrice;
}

// 应用多重优惠（拓展挑战）
function applyMultipleDiscounts(...callbacks) {
    let price = calcTotal();
    console.log(`原总价: ¥${price.toFixed(2)}`);
    
    callbacks.forEach((callback, index) => {
        price = callback(price);
        console.log(`优惠${index + 1}后: ¥${price.toFixed(2)}`);
    });
    
    return price;
}
// 优惠策略1：满200减30（匿名函数）
const full200Minus30 = function(price) {
    return price > 200 ? price - 30 : price;
};

// 优惠策略2：打9折（匿名函数）
const discount90 = function(price) {
    return price * 0.9;
};

// 优惠策略3：满500减100（函数表达式）
const full500Minus100 = function(price) {
    if (price >= 500) {
        return price - 100;
    }
    return price;
};

// 优惠策略4：满1000打85折（函数表达式）
const full1000Discount85 = function(price) {
    if (price >= 1000) {
        return price * 0.85;
    }
    return price;
